#include<bits/stdc++.h>
using namespace std;

int main()
{
    int a[3];
    char b[3];
    cout << "Dia chi 3 phan tu cua a: " << endl;
    for(int i = 0;i < 3; i++ )
    {
        cout << (int)&a[i] << ' ';
    }
    cout << endl << " Dia chi 3 phan tu cua b: " << endl;
    for(int i = 0; i < 3; i++)
    {
        cout << (int)&b[i] << ' ';
    }
    return 0;
}
